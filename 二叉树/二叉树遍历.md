import sys
from collections import deque

class TreeNode:
    def __init__(self, val, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right

class Solution:
    # 前中后序遍历 递归 (当前为后序)
    def recur_1(self, root):
        ans = []
        def recur(node):
            if not node: return
            recur(node.left)
            recur(node.right)
            ans.append(node.val)
        recur(root)
        print(ans)

    # 迭代遍历 (统一标记法 - 后序)
    def iter_2(self, root):
        if not root: return
        ans = []
        stack = [root] # 迭代通常直接用 list 充当栈
        while stack:
            node = stack.pop()
            if node:
                stack.append(node)
                stack.append(None) # 标记
                if node.right: stack.append(node.right)
                if node.left: stack.append(node.left)
            else:
                node = stack.pop()
                ans.append(node.val)
        print(ans)

    # 层序遍历 递归
    def level_recur_2(self, root):
        levels = []
        def level_recur(node, level):
            if not node: return 
            if len(levels) == level:
                levels.append([])
            levels[level].append(node.val)
            level_recur(node.left, level + 1)
            level_recur(node.right, level + 1)
        level_recur(root, 0)
        print(levels)
    
    # 层序遍历 迭代
    def level_iter_2(self, root):
        if not root: return
        que = deque([root])
        levels = []
        while que:
            current_level = []
            for _ in range(len(que)):
                node = que.popleft()
                current_level.append(node.val)
                if node.left: que.append(node.left)
                if node.right: que.append(node.right)
            levels.append(current_level)
        print(levels)

def bulit_tree(inputs):
    if not inputs or inputs[0] == 'null' or inputs[0] == '':
        return None
    
    root = TreeNode(int(inputs[0]))
    que = deque([root])
    i = 1
    while que and i < len(inputs):
        node = que.popleft()

        # 处理左子节点
        if i < len(inputs) and inputs[i] != 'null':
            node.left = TreeNode(int(inputs[i]))
            que.append(node.left)
        i += 1

        # 处理右子节点
        if i < len(inputs) and inputs[i] != 'null':
            node.right = TreeNode(int(inputs[i]))
            que.append(node.right)
        i += 1
    return root

def main():
    # 改进的输入处理：兼容 [1,2,3] 或 1,2,3 格式
    for line in sys.stdin:
        line = line.strip()
        if not line: continue
        # 去掉方括号并分割
        clean_line = line.replace('[','').replace(']','')
        inputs = [da.strip() for da in clean_line.split(',') if da.strip()]

        root = bulit_tree(inputs)
        solution = Solution()
        
        # 你可以切换你想调用的方法
        solution.level_recur_2(root)

if __name__ == '__main__':
    main()
